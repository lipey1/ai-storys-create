# Configurações de ciclo para o app.js
# Adicione estas variáveis ao seu arquivo .env:

# MAX_CYCLES=0          # 0 = ciclos infinitos, ou número específico de ciclos
# CYCLE_DELAY=10000     # Tempo de espera entre ciclos (em milissegundos) - padrão: 10 segundos
# RETRY_DELAY=30000     # Tempo de espera após erro (em milissegundos) - padrão: 30 segundos

# Exemplos:
# MAX_CYCLES=5         # Executa 5 ciclos e para
# MAX_CYCLES=0         # Executa ciclos infinitos até Ctrl+C
# CYCLE_DELAY=5000     # 5 segundos entre ciclos
# RETRY_DELAY=60000    # 1 minuto após erro
